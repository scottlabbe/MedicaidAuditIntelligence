2) Tiny search component that syncs with your router

// client/src/components/HeaderSearch.tsx
import { useState, useEffect } from "react";
import { useLocation } from "wouter";
import { Search } from "lucide-react";

export default function HeaderSearch() {
  const [location, setLocation] = useLocation();
  const [q, setQ] = useState("");

  // preload input from current URL (if user lands on /explore?query=foo)
  useEffect(() => {
    const params = new URLSearchParams(window.location.search);
    setQ(params.get("query") ?? "");
  }, [location]);

  const go = (value: string) => {
    const params = new URLSearchParams(window.location.search);
    if (value) params.set("query", value); else params.delete("query");
    // Clear page param if present to avoid landing mid-pagination
    params.delete("page");
    setLocation(`/explore${params.toString() ? `?${params.toString()}` : ""}`);
  };

  return (
    <div className="relative">
      <Search className="absolute left-3 top-1/2 -translate-y-1/2 h-5 w-5 text-muted-foreground" aria-hidden="true" />
      <input
        value={q}
        onChange={(e) => setQ(e.target.value)}
        onKeyDown={(e) => { if (e.key === "Enter") go(q); }}
        className="w-full rounded-2xl border border-border bg-card/70 pl-11 pr-16 py-3 text-foreground placeholder:text-muted-foreground shadow-sm focus-ring"
        placeholder="Search by keyword, state, agency, or topic…"
        aria-label="Search audit reports"
      />
      <kbd
        className="pointer-events-none absolute right-3 top-1/2 -translate-y-1/2 text-[10px] text-muted-foreground border border-border rounded-md px-1.5 py-0.5 bg-card/60"
        aria-hidden="true"
      >
        ⌘K
      </kbd>
    </div>
  );
}

This respects your color system and utilities (bg-background, text-foreground, border-border, focus-ring, warm shadows) so it fits your existing theme without fighting legacy colors.  ￼
