1) Minimal sticky header (no search, no auth)

// client/src/components/layout/SiteHeader.tsx
import { useLocation } from "wouter";

const NAV = [
  { href: "/explore", label: "Explore" },
  { href: "/reports", label: "Reports" },
  { href: "/about",   label: "About"  },
];

export default function SiteHeader() {
  const [loc] = useLocation();

  return (
    <header className="sticky top-0 z-50 bg-background/70 backdrop-blur border-b border-border">
      <div className="mx-auto max-w-7xl h-16 px-4 sm:px-6 lg:px-8 flex items-center justify-between">
        <a href="/" className="flex items-center gap-2">
          <div className="h-8 w-8 rounded-xl bg-primary warm-shadow" />
          <span className="font-semibold tracking-tight text-foreground">
            Medicaid Audit Explorer
          </span>
        </a>

        <nav className="hidden md:flex items-center gap-6 text-sm">
          {NAV.map(({ href, label }) => {
            const active = loc === href || (href !== "/" && loc.startsWith(href));
            return (
              <a
                key={href}
                href={href}
                aria-current={active ? "page" : undefined}
                className={`transition-colors ${
                  active
                    ? "text-foreground"
                    : "text-muted-foreground hover:text-foreground"
                }`}
              >
                {label}
              </a>
            );
          })}
        </nav>
      </div>
    </header>
  );
}

	•	Uses your theme tokens (bg-background, text-foreground, border-border, bg-primary) so legacy one-offs won’t leak.
	•	Keeps attention on the hero search on Home and the filter panel on Explore.

In button.tsx, make sure the default variant already uses your primary tokens and add a comfortable lg size for hero CTAs. (This matches how you’re styling things elsewhere.)

// button.tsx (shadcn-style cva)
const buttonVariants = cva(

    variants: {

        default: "bg-primary text-primary-foreground hover:opacity-90",
